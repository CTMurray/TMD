import javax.swing.JPanel;
import java.awt.*;
import java.awt.image.*;

/*******************************************************************************
 * Class that extends the JPanel that will house the movie poster
 * 
 * @author Kimberlin Steffens, Raymundo Ramirez, Camaal Murray, Nicolas Heady
 * @version First Release
 ******************************************************************************/
public class BackgroundPanel extends JPanel {

	/** The BufferedImage object that holds the poster */
	private BufferedImage bufferedImage;

	/** The Graphics2D object that manipulates the panel */
	private Graphics2D graphics;

	/** The serialVersionUID. */
	private static final long serialVersionUID = 1L;

	/**
	 *  The constructor for BackgroundPanel
	 */
	public BackgroundPanel() {
		initialBackground();
	}

	/**
	 * Creates the initial background that will be replaced
	 * with the movie poster
	 */
	public void initialBackground() {

		bufferedImage = new BufferedImage(200, 200, BufferedImage.TYPE_INT_ARGB);
		graphics = bufferedImage.createGraphics();
		FontMetrics fontMetrics = graphics.getFontMetrics();
		graphics.setColor(getForeground());

		String text = "Please search for a movie";
		int y = (200 - (fontMetrics.getHeight() * 2)) / 2;
		int x = (200 - fontMetrics.stringWidth(text)) / 2;
		graphics.drawString(text, x, y + fontMetrics.getAscent());
		graphics.dispose();

	}

	/**
	 * The method that will apply the movie poster to
	 * the panel
	 * 
	 * @param background
	 * 				a BufferedImage object that will be displayed
	 */
	public void changeBackground(BufferedImage background) {

		bufferedImage = background;
		graphics = bufferedImage.createGraphics();
		graphics.setColor(getForeground());
		graphics.dispose();

	}

	/**
	 * Method to get the preferred size of the panel
	 * 
	 * @return Dimension 
	 */
	public Dimension getPreferredSize() {
		return new Dimension(350, 500);
	}

	/**
	 * Method that updates the graphics of the panel
	 */
	protected void paintComponent(Graphics g) {
		super.paintComponent(g);
		int x = (getWidth() - bufferedImage.getWidth()) / 2;
		int y = (getHeight() - bufferedImage.getHeight()) / 2;
		g.drawImage(bufferedImage, x, y, this);
	}

}
